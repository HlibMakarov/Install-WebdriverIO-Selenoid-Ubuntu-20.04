// Update

//Install Docker

sudo apt install docker.io -y

//Enable Docker

sudo systemctl start docker
sudo systemctl enable docker

//User access for Docker
sudo usermod -aG docker SOMEUSERNAME // "SOMEUSERNAME" Change for your username

//Install curl

sudo apt install curl


//Install Selenoid http://localhost:4444 

curl -s https://aerokube.com/cm/bash | bash \
&&
sudo ./cm selenoid start --vnc --tmpfs 128

//Selenoid UI install http://localhost:8080/

sudo docker run -d --name selenoid-ui  \
    --link selenoid                 \
    -p 8080:8080                    \
    aerokube/selenoid-ui --selenoid-uri=http://selenoid:4444


sudo docker ps                  //all started containers
sudo docker ps -a               //all containers
sudo docker start 'id_container'     //start container




// in WebdriverIO


mkdir webdriverio-test && cd webdriverio-test
&&
npm init -y
&&
npm i --save-dev @wdio/cli
&&
npx wdio config -y
&&
mkdir -p ./test/specs
&&
touch ./test/specs/basic.js
&&
echo "describe('webdriver.io page', () => {
    it('should open API,Help,Contribute pages', () => {
     browser.url('https://webdriver.io')
     browser.setWindowSize(1920, 1080)
 
     const APIbut = $('li a[href="/docs/api.html"]')
     const Help   = $('li a[href="/help.html"]')
     const Contribute = $('li a[href="/docs/contribute.html"]')
     

     browser.pause(5000)
     APIbut.isDisplayed()
     APIbut.click()
     browser.pause(3000)
     Help.isDisplayed()
     Help.click()
     browser.pause(3000)
     Contribute.isDisplayed()
     Contribute.click()
     browser.pause(3000)
 })
 })" >> ./test/specs/basic.js
&&
npm install wdio-chromedriver-service --save-dev
&&
npm install @wdio/selenium-standalone-service --save-dev
&&
npx wdio wdio.conf.js
